"""
Step 2: Create Pipeline Data Structure and Selector
- Define pipeline dictionary with metadata
- Add pipeline selector
- Display pipeline information
"""

import streamlit as st
import random

st.set_page_config(page_title="Data Pipeline Monitor", page_icon="游댃", layout="wide")

st.title("游댃 Data Pipeline Monitor")

# STEP 1: Create pipeline data structure
# In production, this would come from a database or API
pipelines = {
    "User Data ETL": {
        "status": "Running",
        "progress": 75,
        "records": 15234,
        "description": "Extracts user activity data and loads into analytics warehouse"
    },
    "Sales Analytics": {
        "status": "Completed",
        "progress": 100,
        "records": 45123,
        "description": "Aggregates sales data and generates business metrics"
    },
    "Log Processing": {
        "status": "Failed",
        "progress": 45,
        "records": 8934,
        "description": "Processes application logs and extracts error patterns"
    },
    "Report Generation": {
        "status": "Running",
        "progress": 60,
        "records": 3456,
        "description": "Generates daily reports from processed data"
    }
}

# STEP 2: Add pipeline selector
selected_pipeline = st.selectbox(
    "Select Pipeline",
    list(pipelines.keys()),
    help="Choose a pipeline to monitor"
)

# STEP 3: Get selected pipeline data
pipeline = pipelines[selected_pipeline]

# STEP 4: Display pipeline header
st.subheader(f"游늵 {selected_pipeline}")
st.write(pipeline['description'])

st.divider()

# STEP 5: Display basic pipeline information
st.subheader("Pipeline Details")

col1, col2 = st.columns(2)

with col1:
    st.write("**Status:**", pipeline['status'])
    st.write("**Progress:**", f"{pipeline['progress']}%")

with col2:
    st.write("**Records Processed:**", f"{pipeline['records']:,}")
    st.write("**Runtime:**", f"{random.randint(10, 60)} minutes")

# STEP 6: Add color-coded status indicator
st.subheader("Status Indicator")

status_colors = {
    "Running": "游리 Running",
    "Completed": "游릭 Completed",
    "Failed": "游댮 Failed"
}

status_messages = {
    "Running": "Pipeline is currently processing data...",
    "Completed": "Pipeline completed successfully!",
    "Failed": "Pipeline encountered an error and stopped."
}

status_display = status_colors.get(pipeline['status'], pipeline['status'])
st.info(f"{status_display}: {status_messages.get(pipeline['status'], '')}")

# STEP 7: Show all pipelines overview
with st.expander("游늶 All Pipelines Overview"):
    for name, data in pipelines.items():
        icon = "游릭" if data['status'] == "Completed" else "游리" if data['status'] == "Running" else "游댮"
        st.write(f"{icon} **{name}**: {data['status']} ({data['progress']}%)")

st.divider()
st.caption("Built with Streamlit 游꾿")
