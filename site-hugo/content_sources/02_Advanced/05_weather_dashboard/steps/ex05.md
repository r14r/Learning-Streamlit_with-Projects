"""

- Display the forecast in a column layout
- Create 5 columns, one for each day
- Make the forecast more visually appealing

- st.columns() with multiple columns
- Iterating through columns and data together
- enumerate() for getting index and value
"""

    st.metric("ğŸŒ¡ï¸ Temperature", f"{temp}Â°C")

    forecast_data.append({

        st.write(f"**{data['Day']}**")
