"""
Step 6: Adding Raw JSON Display

In this final step, we'll:
- Add an expander to show raw JSON data
- Display the first few records in JSON format
- Complete the API dashboard

Key Concepts:
- st.expander() for collapsible sections
- st.json() for displaying JSON data
- List slicing to show only first few records
"""

import streamlit as st
import requests
import pandas as pd

st.set_page_config(page_title="API Dashboard", page_icon="üåê", layout="wide")

st.title("üåê API Dashboard")
st.write("Fetch and display data from REST APIs")

# API endpoint selection
api_choice = st.selectbox("Select API Endpoint",
    ["Posts", "Users", "Comments", "Todos", "Albums"])

# Base URL for the API
base_url = "https://jsonplaceholder.typicode.com"

# Fetch button
if st.button("Fetch Data", type="primary"):
    with st.spinner("Fetching data..."):
        try:
            endpoint = api_choice.lower()
            response = requests.get(f"{base_url}/{endpoint}")
            response.raise_for_status()
            data = response.json()

            st.success(f"‚úÖ Fetched {len(data)} records")

            # Convert to DataFrame
            df = pd.DataFrame(data)

            # Create two columns for layout
            col1, col2 = st.columns([2, 1])

            # Left column: Data table
            with col1:
                st.subheader("üìä Data Table")
                st.dataframe(df.head(20), use_container_width=True)

            # Right column: Metrics
            with col2:
                st.subheader("üìã Info")
                st.metric("Total Records", len(df))
                st.metric("Columns", len(df.columns))

            # Raw JSON display in an expander
            with st.expander("üóÇÔ∏è Raw JSON"):
                st.write("Showing first 3 records:")
                st.json(data[:3])

        except Exception as e:
            st.error(f"Error: {str(e)}")

else:
    st.markdown("""
    ### About this app:
    This dashboard fetches data from the JSONPlaceholder API, which is a free fake API for testing and prototyping.

    **Available endpoints:**
    - **Posts**: Blog posts
    - **Users**: User information
    - **Comments**: Comments on posts
    - **Todos**: Todo items
    - **Albums**: Photo albums

    Click the button above to fetch data!
    """)

st.divider()
st.caption("Built with Streamlit üéà")
