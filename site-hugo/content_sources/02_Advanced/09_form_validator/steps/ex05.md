"""
Step 5: Adding Age and Phone Validation

In this step, we'll:
- Add age validation (must be 18+)
- Add phone format validation using regex
- Display submitted data on success

Key Concepts:
- Numeric validation
- Phone number regex pattern
- st.expander() for showing submitted data
- Complete form validation
"""

import streamlit as st
import re

st.set_page_config(page_title="Form Validator", page_icon="‚úîÔ∏è", layout="centered")

st.title("‚úîÔ∏è Form Validator")

with st.form("validation_form"):
    st.subheader("User Registration")

    username = st.text_input("Username (min 4 characters)")
    email = st.text_input("Email")
    password = st.text_input("Password", type="password")
    confirm_password = st.text_input("Confirm Password", type="password")
    age = st.number_input("Age", min_value=0, max_value=120, value=18)
    phone = st.text_input("Phone (format: XXX-XXX-XXXX)")

    submitted = st.form_submit_button("Submit", type="primary")

    if submitted:
        errors = []

        # Username validation
        if len(username) < 4:
            errors.append("Username must be at least 4 characters")

        # Email validation
        if not re.match(r"[^@]+@[^@]+\.[^@]+", email):
            errors.append("Invalid email format")

        # Password validation
        if len(password) < 8:
            errors.append("Password must be at least 8 characters")

        # Password match validation
        if password != confirm_password:
            errors.append("Passwords don't match")

        # Age validation
        if age < 18:
            errors.append("Must be 18 or older")

        # Phone format validation
        if not re.match(r"\d{3}-\d{3}-\d{4}", phone):
            errors.append("Invalid phone format (use XXX-XXX-XXXX)")

        # Display results
        if errors:
            for error in errors:
                st.error(f"‚ùå {error}")
        else:
            st.success("‚úÖ Form validated successfully!")
            with st.expander("Submitted Data"):
                st.write(f"Username: {username}")
                st.write(f"Email: {email}")
                st.write(f"Age: {age}")
                st.write(f"Phone: {phone}")

st.divider()
st.caption("Built with Streamlit üéà")
