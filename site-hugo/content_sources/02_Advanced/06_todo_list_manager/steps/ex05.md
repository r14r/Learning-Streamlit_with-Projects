"""
Step 5: Adding Task Actions (Complete/Delete)

In this step, we'll:
- Add buttons for completing and deleting tasks
- Use unique keys for each button
- Handle task completion and deletion

Key Concepts:
- Button keys with f-strings for uniqueness
- Toggling boolean values
- list.pop() for removing items
- Column layout for action buttons
"""

import streamlit as st
from datetime import datetime

st.set_page_config(page_title="TODO List Manager", page_icon="‚úÖ", layout="centered")

st.title("‚úÖ TODO List Manager")

# Initialize session state
if 'todos' not in st.session_state:
    st.session_state.todos = []

# Add task form
with st.form("add_todo"):
    new_task = st.text_input("New task")
    priority = st.selectbox("Priority", ["Low", "Medium", "High"])
    submitted = st.form_submit_button("Add Task", type="primary")

    if submitted and new_task:
        st.session_state.todos.append({
            'task': new_task,
            'priority': priority,
            'completed': False,
            'created': datetime.now().strftime("%Y-%m-%d %H:%M")
        })
        st.success("Task added!")
        st.rerun()

# Display metrics and tasks
if st.session_state.todos:
    col1, col2, col3 = st.columns(3)
    total = len(st.session_state.todos)
    completed = sum(1 for t in st.session_state.todos if t['completed'])

    with col1:
        st.metric("Total Tasks", total)
    with col2:
        st.metric("Completed", completed)
    with col3:
        st.metric("Remaining", total - completed)

    st.divider()

    # Display todos with action buttons
    for idx, todo in enumerate(st.session_state.todos):
        col1, col2, col3 = st.columns([3, 1, 1])

        with col1:
            # Show task text (strikethrough if completed)
            task_text = f"~~{todo['task']}~~" if todo['completed'] else todo['task']
            st.write(f"{task_text} ({todo['priority']})")

        with col2:
            # Complete/Uncomplete button
            if st.button("‚úì", key=f"complete_{idx}"):
                st.session_state.todos[idx]['completed'] = not todo['completed']
                st.rerun()

        with col3:
            # Delete button
            if st.button("üóëÔ∏è", key=f"delete_{idx}"):
                st.session_state.todos.pop(idx)
                st.rerun()
else:
    st.info("No tasks yet. Add one above!")

st.divider()
st.caption("Built with Streamlit üéà")
