"""
Step 4: Adding Task Metrics

In this step, we'll:
- Add metric cards showing task statistics
- Calculate total, completed, and remaining tasks
- Display metrics in columns

Key Concepts:
- List comprehensions for filtering
- sum() with generator expression for counting
- st.metric() for displaying statistics
"""

import streamlit as st
from datetime import datetime

st.set_page_config(page_title="TODO List Manager", page_icon="âœ…", layout="centered")

st.title("âœ… TODO List Manager")

# Initialize session state
if 'todos' not in st.session_state:
    st.session_state.todos = []

# Add task form
with st.form("add_todo"):
    new_task = st.text_input("New task")
    priority = st.selectbox("Priority", ["Low", "Medium", "High"])
    submitted = st.form_submit_button("Add Task", type="primary")

    if submitted and new_task:
        st.session_state.todos.append({
            'task': new_task,
            'priority': priority,
            'completed': False,
            'created': datetime.now().strftime("%Y-%m-%d %H:%M")
        })
        st.success("Task added!")
        st.rerun()

# Display metrics if there are todos
if st.session_state.todos:
    col1, col2, col3 = st.columns(3)

    total = len(st.session_state.todos)
    completed = sum(1 for t in st.session_state.todos if t['completed'])

    with col1:
        st.metric("Total Tasks", total)
    with col2:
        st.metric("Completed", completed)
    with col3:
        st.metric("Remaining", total - completed)

    st.divider()

    # Display todos
    st.subheader("Your Tasks")
    for todo in st.session_state.todos:
        status = "âœ…" if todo['completed'] else "â¬œ"
        st.write(f"{status} {todo['task']} ({todo['priority']})")
else:
    st.info("No tasks yet. Add one above!")

st.divider()
st.caption("Built with Streamlit ðŸŽˆ")
