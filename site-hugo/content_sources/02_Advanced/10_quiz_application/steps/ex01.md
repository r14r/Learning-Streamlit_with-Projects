"""

In this first step, we'll:
- Set up the page configuration
- Create a list of questions with answers
- Display the questions structure

Key Concepts:
- List of dictionaries for storing quiz data
- Data structure organization
- Understanding the quiz format
"""

import streamlit as st

st.set_page_config(page_title="Quiz Application", page_icon="ðŸŽ¯", layout="centered")

st.title("ðŸŽ¯ Quiz Application")

# Define quiz questions
questions = [
    {"q": "What is 2 + 2?", "options": ["3", "4", "5"], "answer": "4"},
    {"q": "What is the capital of France?", "options": ["London", "Paris", "Berlin"], "answer": "Paris"},
    {"q": "What is 10 * 5?", "options": ["50", "55", "45"], "answer": "50"}
]

st.subheader("Quiz Structure")
st.write(f"Total questions: {len(questions)}")

st.info("""
This quiz app will:
1. Display multiple-choice questions
2. Collect user answers
3. Calculate and display the score
4. Provide feedback based on performance
""")

# Preview questions
with st.expander("Preview Questions"):
    for i, q in enumerate(questions):
        st.write(f"{i+1}. {q['q']}")
        st.write(f"   Options: {', '.join(q['options'])}")
        st.write(f"   Correct answer: {q['answer']}")

st.divider()
st.caption("Built with Streamlit ðŸŽˆ")
