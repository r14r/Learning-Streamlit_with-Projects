"""
Step 4: Adding Game Over Logic
Learning objective: End the game after winning or reaching max attempts
"""

import streamlit as st
import random

st.set_page_config(
    page_title="Number Guessing Game",
    page_icon="ðŸŽ²",
    layout="centered"
)

st.title("ðŸŽ² Number Guessing Game")
st.write("I'm thinking of a number between 1 and 100. Can you guess it?")

# Initialize session state
if 'target_number' not in st.session_state:
    st.session_state.target_number = random.randint(1, 100)
    st.session_state.attempts = 0
    st.session_state.game_over = False  # Track if game has ended

# Display game stats
col1, col2 = st.columns(2)

with col1:
    st.metric("Attempts", st.session_state.attempts)

with col2:
    remaining = max(0, 10 - st.session_state.attempts)
    st.metric("Remaining", remaining)

# Game interface
if not st.session_state.game_over and st.session_state.attempts < 10:
    guess = st.number_input("Enter your guess (1-100):", min_value=1, max_value=100, value=50, step=1)

    if st.button("Submit Guess", type="primary"):
        st.session_state.attempts += 1

        if guess == st.session_state.target_number:
            st.session_state.game_over = True
            st.balloons()  # Celebration animation!
            st.success(f"ðŸŽ‰ Congratulations! You guessed it in {st.session_state.attempts} attempts!")
        elif guess < st.session_state.target_number:
            st.info("ðŸ“ˆ Too low! Try a higher number.")
        else:
            st.info("ðŸ“‰ Too high! Try a lower number.")

        # Check if out of attempts
        if st.session_state.attempts >= 10 and not st.session_state.game_over:
            st.session_state.game_over = True
            st.error(f"ðŸ˜” Game Over! The number was {st.session_state.target_number}")

elif st.session_state.game_over:
    st.info("Game Over! Click 'Play Again' to start a new game.")
