06 Image Gallery ‚Äî App

Level: Beginner Datei: app/app.py
Zurueck zum ProjektStep 6: Add expandable info sections

app.py

"""
Image Gallery - Beginner Project 06

Upload and display images with information.
"""

import streamlit as st
from PIL import Image
import io

st.set_page_config(
    page_title="Image Gallery",
    page_icon="üñºÔ∏è",
    layout="wide"
)

st.title("üñºÔ∏è Image Gallery")
st.write("Upload and view your images")

# File uploader
uploaded_files = st.file_uploader(
    "Choose images...", 
    type=['png', 'jpg', 'jpeg'],
    accept_multiple_files=True
)

if uploaded_files:
    st.success(f"‚úÖ Uploaded {len(uploaded_files)} image(s)")
    
    # Display images in a grid
    cols_per_row = 3
    rows = (len(uploaded_files) + cols_per_row - 1) // cols_per_row
    
    for row in range(rows):
        cols = st.columns(cols_per_row)
        for col_idx in range(cols_per_row):
            img_idx = row * cols_per_row + col_idx
            if img_idx < len(uploaded_files):
                uploaded_file = uploaded_files[img_idx]
                
                with cols[col_idx]:
                    # Open and display image
                    image = Image.open(uploaded_file)
                    st.image(image, caption=uploaded_file.name, use_container_width=True)
                    
                    # Image information
                    with st.expander("‚ÑπÔ∏è Image Info"):
                        st.write(f"**Filename**: {uploaded_file.name}")
                        st.write(f"**Size**: {uploaded_file.size / 1024:.2f} KB")
                        st.write(f"**Dimensions**: {image.size[0]} x {image.size[1]} pixels")
                        st.write(f"**Format**: {image.format}")
                        st.write(f"**Mode**: {image.mode}")

else:
    st.info("üëÜ Upload one or more images to get started")
    
    # Show sample instructions
    st.markdown("""
    ### Supported Formats:
    - PNG
    - JPG/JPEG
    
    ### Features:
    - Upload multiple images at once
    - View image details
    - Grid layout display
    """)

st.divider()
st.caption("Built with Streamlit üéà")
